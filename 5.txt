#include <stdio.h>
#include <stdlib.h>

int main()
{
    int z, temp, j,i;
    printf("Unesite duzinu niza (z<=50):");
    do{
    scanf("%d", &z);
    }while(z>50);
    int niz[z];;
    for(i=0;i<z;i++)
    {
        printf("\nUnesite %d. broj:", i+1);
        scanf("%d", &niz[i]);
    }
    for(i=0;i<z;i++)
    {
        for(j=0;j<z;j++){
        if(niz[j]>niz[j+1])
        {
           temp=niz[j+1];
           niz[j+1]=niz[j];
           niz[j]=temp;
        }
        }
    }
    j=1;
    for(i=0;i<z;i++)
    {
        printf("\n %d. broj u nizu: %d",j,niz[i]);
        j++;
    }

}
