#include <stdio.h>
#include <stdlib.h>

int main()
{
    int z,i,suma=0;
    printf("Unesite duzinu niza (z<=20):");
    do{
    scanf("%d", &z);
    }while(z>20);
    int niz[z];;
    for(i=0;i<z;i++)
    {
        printf("\nUnesite %d. broj:", i+1);
        scanf("%d", &niz[i]);
    }
    for(i=0;i<z;i++)
    {
        suma+=(niz[i]*niz[i]);
    }
    printf("\n Zbir kvadrata elemenata niza je:%d", suma);
}
