#include <stdio.h>
#include <stdlib.h>

int main()
{
    int z,i,g=0;
    float avg,suma=0,j;
    printf("Unesite duzinu niza (z<=20):");
    do{
    scanf("%d", &z);
    }while(z>20);
    int niz[z];;
    for(i=0;i<z;i++)
    {
        printf("\nUnesite %d. broj:", i+1);
        scanf("%d", &niz[i]);
    }
    j=0;
    for(i=0;i<z;i++)
    {
        if(niz[i]%3==0)
        {
            j++;
            suma+=niz[i];
            g++;
        }

    }
    avg=suma/j;
    if(g!=0)
    printf("\n Vrednost je: %f", avg);
    else
        printf("\n Niz ne sadrzi broj deljiv sa 3.");
}
